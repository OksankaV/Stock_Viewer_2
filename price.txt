CREATE TABLE price(
id       INTEGER PRIMARY KEY AUTOINCREMENT,
brand    VARCHAR(30) NOT NULL,
family   VARCHAR(50) NOT NULL,
origin   VARCHAR(30) NOT NULL,
comment  VARCHAR(300),
remain   INTEGER,
moreflag BOOLEAN,
supplier VARCHAR(30),
sp       DECIMAL(5,2),
spc      CHAR(3),
sourcestring VARCHAR(400),
minimalorder INTEGER,
deliverytyme INTEGER,
suppliercomment VARCHAR(300),
actualdate DATE,
runflat  BOOLEAN,
sidewall VARCHAR(5),
productiondate VARCHAR(5),
diameterc    VARCHAR(4),
application  VARCHAR(30),
season       VARCHAR(30),
traileraxle  BOOLEAN,
steeringaxle BOOLEAN,
driveaxle    BOOLEAN,
dimensiontype VARCHAR(30),
sectionsize  VARCHAR(30),
bp           DECIMAL(5,2),
bpvat        DECIMAL(5,2),
bppe         DECIMAL(5,2),
rp           DECIMAL(5,2),
rpvat        DECIMAL(5,2),
rppe         DECIMAL(5,2)
);
